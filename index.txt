import React, { useState, useEffect } from 'react';
import { 
  Plane, 
  MapPin, 
  Calendar, 
  DollarSign, 
  Plus, 
  Trash2, 
  Users, 
  CheckCircle2, 
  ChevronRight,
  Clock,
  Compass,
  Wallet,
  ExternalLink,
  RefreshCw,
  Share2,
  AlertCircle
} from 'lucide-react';

/**
 * GOOGLE SHEETS BACKEND CONNECTED
 * URL: https://script.google.com/macros/s/AKfycbzR0z8w4vut1oWjNVoIx3c6VuEAybJD6biDhCV12J3XHX3oZKMhQKbTDzLgZLsG5Nwd/exec
 */

const SCRIPT_URL = "https://script.google.com/macros/s/AKfycbzR0z8w4vut1oWjNVoIx3c6VuEAybJD6biDhCV12J3XHX3oZKMhQKbTDzLgZLsG5Nwd/exec";

const App = () => {
  const [trips, setTrips] = useState([]);
  const [activeTrip, setActiveTrip] = useState(null);
  const [syncing, setSyncing] = useState(false);
  const [error, setError] = useState(null);
  const [view, setView] = useState('dashboard');
  
  // Form States
  const [newTripName, setNewTripName] = useState('');
  const [itineraryItem, setItineraryItem] = useState({ title: '', time: '', date: '', cost: 0 });

  const fetchData = async () => {
    if (!SCRIPT_URL) return;
    setSyncing(true);
    setError(null);
    try {
      const response = await fetch(SCRIPT_URL);
      if (!response.ok) throw new Error('Failed to reach Google Sheets');
      const data = await response.json();
      setTrips(Array.isArray(data) ? data : []);
      
      if (activeTrip) {
        const updated = data.find(t => t.id === activeTrip.id);
        if (updated) setActiveTrip(updated);
      }
    } catch (err) {
      console.error("Fetch error:", err);
      setError("Connection issue. Please ensure your Apps Script is deployed to 'Anyone'.");
    } finally {
      setSyncing(false);
    }
  };

  useEffect(() => {
    fetchData();
    const interval = setInterval(fetchData, 15000); // Poll every 15s for group collaboration
    return () => clearInterval(interval);
  }, [activeTrip?.id]);

  const syncToSheets = async (updatedTrips) => {
    if (!SCRIPT_URL) return;
    setSyncing(true);
    try {
      // Use 'no-cors' mode for POSTing to Apps Script Web Apps
      await fetch(SCRIPT_URL, {
        method: 'POST',
        mode: 'no-cors',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(updatedTrips)
      });
      // Small delay then refresh local state to confirm sync
      setTimeout(fetchData, 1500); 
    } catch (err) {
      console.error("Sync error:", err);
      setError("Failed to save changes to the sheet.");
    } finally {
      setSyncing(false);
    }
  };

  const createTrip = async (e) => {
    e.preventDefault();
    if (!newTripName.trim()) return;

    const newTrip = {
      id: crypto.randomUUID(),
      name: newTripName,
      createdAt: Date.now(),
      itinerary: [],
      budget: 0
    };
    
    const updated = [...trips, newTrip];
    setTrips(updated);
    await syncToSheets(updated);
    setNewTripName('');
  };

  const addItineraryItem = async () => {
    if (!itineraryItem.title || !activeTrip) return;
    
    const newItem = { ...itineraryItem, id: crypto.randomUUID() };
    const updatedItinerary = [...(activeTrip.itinerary || []), newItem];
    const totalBudget = updatedItinerary.reduce((sum, item) => sum + (parseFloat(item.cost) || 0), 0);

    const updatedTrips = trips.map(t => 
      t.id === activeTrip.id 
      ? { ...t, itinerary: updatedItinerary, budget: totalBudget } 
      : t
    );

    setTrips(updatedTrips);
    await syncToSheets(updatedTrips);
    setItineraryItem({ title: '', time: '', date: '', cost: 0 });
  };

  const removeItineraryItem = async (itemId) => {
    const updatedItinerary = activeTrip.itinerary.filter(item => item.id !== itemId);
    const totalBudget = updatedItinerary.reduce((sum, item) => sum + (parseFloat(item.cost) || 0), 0);
    
    const updatedTrips = trips.map(t => 
      t.id === activeTrip.id 
      ? { ...t, itinerary: updatedItinerary, budget: totalBudget } 
      : t
    );

    setTrips(updatedTrips);
    await syncToSheets(updatedTrips);
  };

  const deleteTrip = async (e, id) => {
    e.stopPropagation();
    const updated = trips.filter(t => t.id !== id);
    setTrips(updated);
    await syncToSheets(updated);
  };

  return (
    <div className="min-h-screen bg-slate-50 text-slate-900 font-sans">
      {/* Navigation */}
      <nav className="bg-white border-b border-slate-200 px-6 py-4 flex items-center justify-between sticky top-0 z-10">
        <div 
          className="flex items-center gap-2 cursor-pointer" 
          onClick={() => { setView('dashboard'); setActiveTrip(null); }}
        >
          <div className="bg-emerald-600 p-2 rounded-lg text-white shadow-lg shadow-emerald-100">
            <Plane size={20} />
          </div>
          <h1 className="text-xl font-bold tracking-tight text-slate-800">SheetTravel</h1>
        </div>
        <div className="flex items-center gap-4">
          {error && (
            <div className="hidden md:flex items-center gap-1 text-rose-500 text-xs font-medium bg-rose-50 px-3 py-1 rounded-full border border-rose-100">
              <AlertCircle size={14}/> {error}
            </div>
          )}
          {syncing && (
            <div className="flex items-center gap-2 text-emerald-600 text-xs font-bold bg-emerald-50 px-3 py-1 rounded-full animate-pulse">
              <RefreshCw size={14} className="animate-spin" /> Syncing...
            </div>
          )}
          <div className="bg-slate-100 text-slate-600 px-3 py-1 rounded-full text-xs font-bold flex items-center gap-1 border border-slate-200">
            <div className="w-1.5 h-1.5 bg-emerald-500 rounded-full"></div>
            Google Sheets Connected
          </div>
        </div>
      </nav>

      <main className="max-w-5xl mx-auto p-6">
        {view === 'dashboard' ? (
          <div className="space-y-8 animate-in fade-in duration-500">
            <header className="flex flex-col md:flex-row md:items-end justify-between gap-4">
              <div>
                <h2 className="text-3xl font-extrabold text-slate-900">Your Adventures</h2>
                <p className="text-slate-500 mt-1">Managed collaboratively via your private Google Sheet.</p>
              </div>
              <form onSubmit={createTrip} className="flex gap-2">
                <input 
                  type="text" 
                  placeholder="New Trip Name" 
                  className="px-4 py-2 rounded-xl border border-slate-200 focus:outline-none focus:ring-2 focus:ring-emerald-500 w-full md:w-64 bg-white shadow-sm"
                  value={newTripName}
                  onChange={(e) => setNewTripName(e.target.value)}
                />
                <button type="submit" className="bg-emerald-600 hover:bg-emerald-700 text-white px-5 py-2 rounded-xl flex items-center gap-2 transition-all font-semibold shadow-lg shadow-emerald-100/50 active:scale-95">
                  <Plus size={18} /> New Trip
                </button>
              </form>
            </header>

            <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-6">
              {trips.length === 0 && !syncing ? (
                <div className="col-span-full py-24 text-center border-2 border-dashed border-slate-200 rounded-3xl bg-white/50 backdrop-blur-sm">
                  <Compass size={56} className="mx-auto text-slate-300 mb-4" />
                  <p className="text-slate-400 font-medium">No trips found in your sheet.</p>
                  <p className="text-slate-400 text-sm">Create one above to get started!</p>
                </div>
              ) : (
                trips.map(trip => (
                  <div 
                    key={trip.id} 
                    onClick={() => { setActiveTrip(trip); setView('trip-details'); }}
                    className="group bg-white p-6 rounded-3xl border border-slate-200 hover:shadow-xl hover:border-emerald-100 transition-all cursor-pointer relative overflow-hidden active:scale-[0.98]"
                  >
                    <div className="relative z-10">
                      <div className="flex justify-between items-start mb-4">
                        <div className="bg-emerald-50 text-emerald-600 p-3 rounded-2xl">
                          <MapPin size={24} />
                        </div>
                        <button 
                          onClick={(e) => deleteTrip(e, trip.id)}
                          className="text-slate-300 hover:text-rose-500 p-2 transition-colors rounded-lg hover:bg-rose-50"
                        >
                          <Trash2 size={16} />
                        </button>
                      </div>
                      <h3 className="text-xl font-bold text-slate-800 mb-1 group-hover:text-emerald-600 transition-colors">
                        {trip.name}
                      </h3>
                      <div className="flex items-center gap-4 mt-4 pt-4 border-t border-slate-50">
                        <div className="flex items-center gap-1 text-slate-500 text-sm font-medium">
                          <Wallet size={14} className="text-emerald-500" />
                          <span>${trip.budget || 0}</span>
                        </div>
                        <div className="flex items-center gap-1 text-slate-500 text-sm font-medium">
                          <Calendar size={14} className="text-emerald-500" />
                          <span>{trip.itinerary?.length || 0} activities</span>
                        </div>
                      </div>
                    </div>
                  </div>
                ))
              )}
            </div>
          </div>
        ) : (
          <div className="animate-in slide-in-from-right duration-300">
            <button 
              onClick={() => setView('dashboard')}
              className="text-slate-500 hover:text-emerald-600 flex items-center gap-1 mb-6 font-medium bg-white px-4 py-2 rounded-xl border border-slate-100 transition-all shadow-sm hover:shadow-md"
            >
              <ChevronRight size={18} className="rotate-180" /> Back to Dashboard
            </button>

            <div className="grid grid-cols-1 lg:grid-cols-3 gap-8">
              <div className="lg:col-span-2 space-y-6">
                <div className="bg-white p-8 rounded-3xl border border-slate-200 shadow-sm relative overflow-hidden">
                  <div className="absolute top-0 left-0 w-full h-2 bg-emerald-600"></div>
                  <div className="flex items-center justify-between mb-8">
                    <h2 className="text-3xl font-black text-slate-900">{activeTrip?.name}</h2>
                    <div className="bg-emerald-600 text-white px-5 py-2 rounded-2xl flex items-center gap-2 font-bold shadow-lg shadow-emerald-100/50">
                      <DollarSign size={18} /> {activeTrip?.budget} Total
                    </div>
                  </div>

                  <div className="bg-slate-50 p-6 rounded-2xl mb-8 border border-slate-100">
                    <h4 className="text-xs font-bold text-slate-400 uppercase tracking-widest mb-4 flex items-center gap-2">
                       <Plus size={14} /> New Activity
                    </h4>
                    <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
                      <input 
                        type="text" 
                        placeholder="What's the plan?" 
                        className="col-span-1 md:col-span-2 px-4 py-3 rounded-xl border border-slate-200 focus:ring-2 focus:ring-emerald-500 focus:outline-none"
                        value={itineraryItem.title}
                        onChange={(e) => setItineraryItem({...itineraryItem, title: e.target.value})}
                      />
                      <input 
                        type="date" 
                        className="px-4 py-3 rounded-xl border border-slate-200 focus:outline-none focus:ring-1 focus:ring-emerald-500"
                        value={itineraryItem.date}
                        onChange={(e) => setItineraryItem({...itineraryItem, date: e.target.value})}
                      />
                      <input 
                        type="time" 
                        className="px-4 py-3 rounded-xl border border-slate-200 focus:outline-none focus:ring-1 focus:ring-emerald-500"
                        value={itineraryItem.time}
                        onChange={(e) => setItineraryItem({...itineraryItem, time: e.target.value})}
                      />
                      <div className="relative">
                        <DollarSign size={16} className="absolute left-4 top-1/2 -translate-y-1/2 text-slate-400" />
                        <input 
                          type="number" 
                          placeholder="Cost" 
                          className="w-full pl-10 pr-4 py-3 rounded-xl border border-slate-200 focus:outline-none focus:ring-1 focus:ring-emerald-500"
                          value={itineraryItem.cost}
                          onChange={(e) => setItineraryItem({...itineraryItem, cost: e.target.value})}
                        />
                      </div>
                      <button 
                        onClick={addItineraryItem}
                        disabled={syncing}
                        className="bg-emerald-600 hover:bg-emerald-700 disabled:bg-slate-300 text-white font-bold py-3 rounded-xl transition-all shadow-lg shadow-emerald-50 flex items-center justify-center gap-2 active:scale-95"
                      >
                        {syncing ? <RefreshCw size={18} className="animate-spin" /> : 'Add to Schedule'}
                      </button>
                    </div>
                  </div>

                  <div className="space-y-4">
                    <h4 className="text-xs font-bold text-slate-400 uppercase tracking-widest">Planned Events</h4>
                    {!activeTrip?.itinerary?.length ? (
                      <div className="text-center py-16 bg-slate-50/50 rounded-2xl border-2 border-dotted border-slate-200">
                        <p className="text-slate-400 italic">Itinerary is currently empty.</p>
                      </div>
                    ) : (
                      activeTrip.itinerary.map((item) => (
                        <div key={item.id} className="flex items-center gap-4 p-5 rounded-2xl border border-slate-100 hover:bg-emerald-50/20 transition-all group bg-white shadow-sm hover:shadow-md">
                          <div className="bg-emerald-50 p-3 rounded-xl text-emerald-600">
                            <Clock size={20} />
                          </div>
                          <div className="flex-1">
                            <h5 className="font-bold text-slate-800">{item.title}</h5>
                            <div className="flex flex-wrap items-center gap-3 text-xs text-slate-500 mt-1">
                              <span className="flex items-center gap-1 bg-slate-100 px-2 py-1 rounded-md"><Calendar size={12} /> {item.date || 'No Date'}</span>
                              <span className="flex items-center gap-1 bg-slate-100 px-2 py-1 rounded-md"><Clock size={12} /> {item.time || '--:--'}</span>
                              <span className="font-bold text-emerald-600 ml-auto text-sm bg-emerald-50 px-2 py-1 rounded-lg">${item.cost || 0}</span>
                            </div>
                          </div>
                          <button 
                            onClick={() => removeItineraryItem(item.id)}
                            className="opacity-0 group-hover:opacity-100 p-2 text-slate-300 hover:text-rose-500 hover:bg-rose-50 rounded-lg transition-all"
                          >
                            <Trash2 size={18} />
                          </button>
                        </div>
                      ))
                    )}
                  </div>
                </div>
              </div>

              <div className="space-y-6">
                <div className="bg-white p-6 rounded-3xl border border-slate-200 shadow-sm">
                  <h3 className="font-bold text-slate-800 mb-4 flex items-center gap-2">
                    <Share2 size={20} className="text-emerald-600" /> Crew Sharing
                  </h3>
                  <div className="space-y-4">
                    <p className="text-xs text-slate-500 leading-relaxed">
                      Anyone with this app URL can view and edit these trips. Changes sync to your Google Sheet in real-time.
                    </p>
                    <div className="p-3 bg-slate-50 rounded-xl text-[10px] font-mono break-all text-slate-400 border border-slate-100 select-all">
                      {window.location.href}
                    </div>
                  </div>
                </div>

                <div className="bg-slate-900 p-6 rounded-3xl text-white shadow-xl">
                  <div className="flex items-center gap-2 mb-4">
                    <div className="w-2 h-2 bg-emerald-400 rounded-full animate-pulse"></div>
                    <span className="font-bold text-sm">Live Sheet Sync</span>
                  </div>
                  <p className="text-slate-400 text-xs mb-6">
                    All trip data is encrypted as JSON and stored in Cell A1. Do not edit that cell manually to prevent data corruption.
                  </p>
                  <div className="flex items-center justify-between pt-4 border-t border-slate-800">
                    <Users size={16} className="text-slate-600" />
                    <span className="text-xs text-slate-500 font-medium tracking-widest uppercase">Multiplayer Enabled</span>
                  </div>
                </div>
              </div>
            </div>
          </div>
        )}
      </main>
    </div>
  );
};

export default App;
