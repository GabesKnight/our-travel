    <script type="module">
        import { initializeApp } from "https://www.gstatic.com/firebasejs/11.1.0/firebase-app.js";
        import { getFirestore, doc, onSnapshot, setDoc } from "https://www.gstatic.com/firebasejs/11.1.0/firebase-firestore.js";
        import { getAuth, signInAnonymously, onAuthStateChanged, signInWithCustomToken } from "https://www.gstatic.com/firebasejs/11.1.0/firebase-auth.js";

        const firebaseConfig = JSON.parse(__firebase_config);
        const app = initializeApp(firebaseConfig);
        const db = getFirestore(app);
        const auth = getAuth(app);
        const appId = typeof __app_id !== 'undefined' ? __app_id : 'holiday-planner';

        let tripId = new URLSearchParams(window.location.search).get('trip');
        if (!tripId) {
            tripId = crypto.randomUUID();
            const newUrl = new URL(window.location);
            newUrl.searchParams.set('trip', tripId);
            window.history.pushState({}, '', newUrl);
        }

        let isUpdatingFromServer = false;
        let currentUser = null;

        window.addEventListener('DOMContentLoaded', async () => {
            const initAuth = async () => {
                if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {
                    await signInWithCustomToken(auth, __initial_auth_token);
                } else {
                    await signInAnonymously(auth);
                }
            };
            await initAuth();
            onAuthStateChanged(auth, (user) => {
                currentUser = user;
                if (user) setupFirestoreListener();
            });

            document.getElementById('itinerary-content').addEventListener('input', debounce(() => {
                if (!isUpdatingFromServer) saveToFirebase();
            }, 1000));
        });

        function setupFirestoreListener() {
            if (!currentUser) return;
            const tripDocRef = doc(db, 'artifacts', appId, 'public', 'data', 'trips', tripId);
            onSnapshot(tripDocRef, (docSnap) => {
                if (docSnap.exists()) {
                    isUpdatingFromServer = true;
                    renderData(docSnap.data());
                    isUpdatingFromServer = false;
                } else {
                    saveToFirebase(); 
                }
            }, (error) => console.error("Firestore Error:", error));
        }

        async function saveToFirebase() {
            if (!currentUser) return;
            const data = {
                title: document.getElementById('trip-title').value,
                dates: document.getElementById('trip-dates').value,
                days: []
            };
            document.querySelectorAll('.day-group').forEach(group => {
                const day = {
                    headline: group.querySelector('.day-headline').value,
                    date: group.querySelector('.day-date').value,
                    activities: []
                };
                group.querySelectorAll('.activity-item').forEach(item => {
                    day.activities.push({
                        title: item.querySelector('.act-title').value,
                        url: item.querySelector('.act-url').value,
                        notes: item.querySelector('.act-notes').value
                    });
                });
                data.days.push(day);
            });
            const tripDocRef = doc(db, 'artifacts', appId, 'public', 'data', 'trips', tripId);
            await setDoc(tripDocRef, data).catch(err => console.error("Save Error:", err));
        }

        function renderData(data) {
            document.getElementById('trip-title').value = data.title || "";
            document.getElementById('trip-dates').value = data.dates || "";
            const container = document.getElementById('days-container');
            container.innerHTML = "";
            if (!data.days || data.days.length === 0) {
                document.getElementById('empty-state').classList.remove('hidden');
                return;
            }
            document.getElementById('empty-state').classList.add('hidden');
            data.days.forEach((day, index) => {
                const dayTemplate = document.getElementById('day-template');
                const dayClone = dayTemplate.content.cloneNode(true);
                dayClone.querySelector('.day-number').textContent = index + 1;
                dayClone.querySelector('.day-headline').value = day.headline || "";
                dayClone.querySelector('.day-date').value = day.date || "";
                const actContainer = dayClone.querySelector('.activities-container');
                (day.activities || []).forEach(act => {
                    const actTemplate = document.getElementById('activity-template');
                    const actClone = actTemplate.content.cloneNode(true);
                    actClone.querySelector('.act-title').value = act.title || "";
                    actClone.querySelector('.act-url').value = act.url || "";
                    actClone.querySelector('.act-notes').value = act.notes || "";
                    actContainer.appendChild(actClone);
                });
                container.appendChild(dayClone);
            });
        }

        function debounce(func, timeout = 300) {
            let timer;
            return (...args) => {
                clearTimeout(timer);
                timer = setTimeout(() => { func.apply(this, args); }, timeout);
            };
        }

        window.addDay = () => {
            const container = document.getElementById('days-container');
            const template = document.getElementById('day-template');
            container.appendChild(template.content.cloneNode(true));
            renderDataFromDOM(); // Refresh numbers
            saveToFirebase();
        };

        function renderDataFromDOM() {
            document.querySelectorAll('.day-number').forEach((span, i) => span.textContent = i + 1);
        }

        window.removeDay = (btn) => {
            btn.closest('.day-group').remove();
            renderDataFromDOM();
            saveToFirebase();
        };

        window.addActivity = (btn) => {
            const container = btn.closest('.day-group').querySelector('.activities-container');
            container.appendChild(document.getElementById('activity-template').content.cloneNode(true));
            saveToFirebase();
        };

        window.removeActivity = (btn) => {
            btn.closest('.activity-item').remove();
            saveToFirebase();
        };

        window.copyShareLink = () => {
            const url = window.location.href;
            const tempInput = document.createElement("input");
            document.body.appendChild(tempInput);
            tempInput.value = url;
            tempInput.select();
            document.execCommand("copy");
            document.body.removeChild(tempInput);
            const btn = document.getElementById('share-btn');
            const originalText = btn.innerHTML;
            btn.innerHTML = "🔗 Link Copied";
            setTimeout(() => btn.innerHTML = originalText, 2000);
        };
    </script>

        async function exportToPDF() {
            const { jsPDF } = window.jspdf;
            const element = document.getElementById('itinerary-content');
            const title = document.getElementById('trip-title').value || 'My Itinerary';
            const btn = event.currentTarget;
            
            btn.disabled = true;
            const originalHTML = btn.innerHTML;
            btn.innerHTML = `<span class="animate-pulse flex items-center gap-2">Generating PDF...</span>`;

            try {
                const canvas = await html2canvas(element, {
                    scale: 2,
                    useCORS: true,
                    backgroundColor: '#ffffff',
                    onclone: (clonedDoc) => {
                        clonedDoc.querySelectorAll('input, textarea').forEach(input => {
                            const p = clonedDoc.createElement('div');
                            let val = input.value;
                            
                            if (input.type === 'date' && val) {
                                const d = new Date(val);
                                val = d.toLocaleDateString('en-US', { weekday: 'short', day: 'numeric', month: 'short', year: 'numeric' });
                            }
                            
                            p.innerText = val || "";
                            
                            const isTitle = input.id === 'trip-title';
                            const isHeadline = input.classList.contains('day-headline');
                            const isActivityTitle = input.classList.contains('act-title');
                            
                            p.style.cssText = `
                                font-family: ${isTitle || isHeadline ? "'Playfair Display', serif" : "'Inter', sans-serif"};
                                font-size: ${isTitle ? '48px' : isHeadline ? '28px' : isActivityTitle ? '20px' : '14px'};
                                font-weight: ${isTitle || isHeadline || isActivityTitle ? 'bold' : 'normal'};
                                color: ${isTitle ? '#ffffff' : '#1e293b'};
                                margin-bottom: 8px;
                                white-space: pre-wrap;
                            `;
                            input.parentNode.replaceChild(p, input);
                        });
                        clonedDoc.querySelectorAll('.no-print').forEach(el => el.style.display = 'none');
                    }
                });

                const imgData = canvas.toDataURL('image/jpeg', 0.98);
                const pdf = new jsPDF('p', 'mm', 'a4');
                const pdfWidth = pdf.internal.pageSize.getWidth();
                const pdfHeight = (canvas.height * pdfWidth) / canvas.width;
                
                pdf.addImage(imgData, 'JPEG', 0, 0, pdfWidth, pdfHeight);
                pdf.save(`${title.replace(/\s+/g, '_')}_Itinerary.pdf`);
            } catch (err) {
                console.error("PDF Export Error:", err);
            } finally {
                btn.disabled = false;
                btn.innerHTML = originalHTML;
            }
        }
    </script>
